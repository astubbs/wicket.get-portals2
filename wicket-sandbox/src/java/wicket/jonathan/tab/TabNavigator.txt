package wicket.markup.html.jontab;

import java.awt.Graphics2D;
import java.util.ArrayList;
import java.util.List;

import wicket.markup.html.image.resource.DynamicImageResource;

/**
 * Draws a basic tab component, varying the image depending on which page is active.
 * 
 * @author Jonathan Locke
 */
public final class TabNavigator extends DynamicImageResource
{
	/** Serial Version ID. */
	private static final long serialVersionUID = 209001445308790198L;
	
	/** List of Tab objects in this navigator */
	private List tabs = new ArrayList();

	/**
	 * @see DynamicImageResource#DynamicImage(String, int, int)
	 */
	public TabNavigator(final String name, final int width, final int height)
	{
		super(name, width, height);
	}
	
	/**
	 * @param tab Tab to add
	 */
	public void add(Tab tab)
	{
		tabs.add(tab);
	}

	/**
	 * @see wicket.markup.html.image.DynamicImageResource#render(java.awt.Graphics2D)
	 */
	protected void render(Graphics2D graphics)
	{
		
	}
}
