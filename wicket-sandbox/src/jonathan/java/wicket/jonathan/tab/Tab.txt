package wicket.markup.html.jontab;

import wicket.Page;

/**
 * Definition of a tab within a TabNavigator component.
 * 
 * @author Jonathan Locke
 */
public abstract class Tab
{
	/** Serial Version ID. */
	private static final long serialVersionUID = 209001445308790198L;
	
	/** Label on tab */
	final String label;

	/**
	 * Constructor
	 * @param label Label for tab
	 */
	public Tab(final String label)
	{
		this.label = label;
	}
	
	protected abstract boolean isActive(final Page page);
}
